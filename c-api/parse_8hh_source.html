80 auto process_path_(std::filesystem::path path, bool required) -> bool {
81 if (std::filesystem::exists(path)) {
82 path = std::filesystem::canonical(path);
83 auto rel = path.lexically_relative(root_);
84 if (!std::filesystem::is_directory(path)) {
85 if (seen_.emplace(path).second) {
86 fin_.open(rel);
87 parser_.init(fin_, *store_->string(rel.c_str()));
88 process_(path.parent_path());
89 } else {

91 }
92 } else {
93 GRINGO_REPORT(*log_, error) << "cannot include directory: " << rel;
94 parse_error_ = true;
95 }
96 return true;
97 }
98 if (required) {

100 parse_error_ = true;

107 void process_(std::filesystem::path const &dir) {
108 prg_->ensure_base();
109 while (true) {
110 auto [stm, res] = parser_.scan();
111 parse_error_ = parse_error_ || !res;
112 if (!stm) {
113 fin_.close();
114 break;
115 }
116 if (auto *include = std::get_if<Input::StmInclude>(&*stm); include != nullptr) {
117 includes_.emplace_back(dir, *include);
118 } else {
119 if (auto *script = std::get_if<Input::StmScript>(&*stm); exec_ != nullptr && script != nullptr) {
120 exec_->exec(script->loc(), *log_, script->type().view(), script->value().view());
121 }
122 prg_->add(*store_, *std::move(stm));
123 }
124 }
125 }
