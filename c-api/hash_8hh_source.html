173inline auto hash_combine(size_t seed, size_t h) -> size_t {
174 // NOLINTBEGIN(readability-magic-numbers)
175 if constexpr (sizeof(size_t) == sizeof(uint64_t)) {
176 seed *= 0x87c37b91114253d5;
177 seed = (seed >> 31) | (seed << 33);
178 seed *= 0x4cf5ad432745937f;
179 seed ^= h;
180 seed = (seed >> 27) | (seed << 37);
181 seed = seed * 5 + 0x52dce729;
182 } else {
183 seed *= 0xcc9e2d51;
184 seed = (seed >> 17) | (seed << 15);
185 seed *= 0x1b873593;
186 seed ^= h;
187 seed = (seed >> 19) | (seed << 13);
188 seed = seed * 5 + 0xe6546b64;
189 }
190 return seed;
191 // NOLINTEND(readability-magic-numbers)
192}

200inline auto hash_mix(size_t h) -> size_t {
201 // NOLINTBEGIN(readability-magic-numbers)
202 if constexpr (sizeof(size_t) == sizeof(uint64_t)) {
203 h ^= h >> 33;
204 h *= 0xff51afd7ed558ccd;
205 h ^= h >> 33;
206 h *= 0xc4ceb9fe1a85ec53;
207 h ^= h >> 33;
208 } else {
209 h ^= h >> 16;
210 h *= 0x85ebca6b;
211 h ^= h >> 13;
212 h *= 0xc2b2ae35;
213 h ^= h >> 16;
214 }
215 return h;
216 // NOLINTEND(readability-magic-numbers)
217}

221template <class T> auto value_hash_range(T const &x) -> size_t {
222 // NOLINTBEGIN(readability-magic-numbers)
223 // note: we unroll the loop for small sizes, this compiles very well and leads to noticable speedups
224 // length 8 should cover typical sizes for argument tuples etc. in clingo
225 auto small = []<std::size_t... n>(std::index_sequence<n...>, auto const &x) {
226 return Util::hash_combine(Util::value_hash(x[n])...);
227 };
228 switch (x.size()) {
229 case 0:
230 return small(std::make_index_sequence<0>(), x);
231 case 1:
232 return small(std::make_index_sequence<1>(), x);
233 case 2:
234 return small(std::make_index_sequence<2>(), x);
235 case 3:
236 return small(std::make_index_sequence<3>(), x);
237 case 4:
238 return small(std::make_index_sequence<4>(), x);
239 case 5:
240 return small(std::make_index_sequence<5>(), x);
241 case 6:
242 return small(std::make_index_sequence<6>(), x);
243 case 7:
244 return small(std::make_index_sequence<7>(), x);
245 case 8:
246 return small(std::make_index_sequence<8>(), x);
247 default:
248 return std::accumulate(
249 x.begin() + 1, x.end(), Util::value_hash(*x.begin()),
250 [](auto const &seed, auto const &x) { return Util::hash_combine(seed, Util::value_hash(x)); });
251 }
252 // NOLINTEND(readability-magic-numbers)
253}

257template <class T> auto value_hash(T const &x) -> size_t {
258 if constexpr (requires { x.hash(); }) {
259 return x.hash();

261 return hash_mix(std::hash<T>{}(x));
