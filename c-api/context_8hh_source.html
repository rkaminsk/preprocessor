40 [[nodiscard]] auto add_base(std::tuple<String, size_t, bool> sig) -> BaseMap::iterator {
41 auto aux = std::get<0>(sig).starts_with("#");
42 auto dom_it = (aux ? aux_base_ : atom_base_)->try_emplace(std::move(sig), nullptr).first;
43 if (dom_it->second == nullptr) {
44 dom_it.value() = std::make_unique<Ground::Base>();
45 }
46 return dom_it;
47 }

50 [[nodiscard]] auto add_project(SymbolStore &store, Ground::UTerm const &term,
51 Ground::Base &base) -> std::pair<Ground::UTerm, Ground::LitProject::State *> {
52 size_t vars = 0;
53 auto [it, ins] =
54 project_base_->try_emplace(term->rename(store, Ground::RenameMode::rename_vars, nullptr, &vars));
55 auto const &p_key = *it->first;
56 auto &state = it.value();
57 if (ins) {
58 auto p_name = store.string_ref("#p_" + std::to_string(project_base_->size()));
59 auto p_head = p_key.rename(store, Ground::RenameMode::drop_projection, &p_name, nullptr);
60 auto p_body = p_key.rename(store, Ground::RenameMode::rename_projection, nullptr, &vars);
61 state =
62 std::make_unique<Ground::LitProject::State>(p_name, vars, base, std::move(p_head), std::move(p_body));
63 }
64 return {term->rename(store, Ground::RenameMode::drop_projection, &state->name(), nullptr), state.get()};
65 }
66
67 private:
68 BaseMap *atom_base_;
69 BaseMap *aux_base_;
70 ProjectMap *project_base_;
71};

181 template <class F> void with_simple_lit(Input::Term const &term, F &&fun) {
182 auto provides = std::vector<size_t>{};
183 auto sig = signature(term);
184 assert(sig);
185 auto dom_it = add_base(*sig);
186 auto &base = *dom_it->second;
187 if (auto it = def_map_->find(&term); it != def_map_->end()) {
188 provides = it->second;
189 }
190 std::invoke(std::forward<F>(fun), *sig, build_term(*var_map_, term), base, std::move(provides));
191 }
192
194 template <class F> void with_simple_lit(Input::Lit const &lit, F &&fun) {
195 std::visit(
196 [&]<class T>(T const &lit) {
197 if constexpr (Util::matches<T, Input::LitSymbolic>) {
198 assert(lit.sign() == Sign::none);
199 with_simple_lit(lit.term(), std::forward<F>(fun));
200 return;
201 } else if constexpr (Util::matches<T, Input::LitBool>) {
202 if (!lit.value()) {
203 return;
204 }
205 }
206 throw std::runtime_error("unexpected literal in rule head");
207 },
208 lit);
209 }
