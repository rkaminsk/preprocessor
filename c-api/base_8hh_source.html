101 void update(size_t generation, size_t size) {
102 // initialize the domain
103 // (all atoms are marked as new)
104 if (generation == 0) {
105 generation_ = 0;
106 old_offset_ = 0;
107 all_offset_ = size;
108 }
109 // the generation has been incremented by one
110 // (freshly added atoms are marked new)
111 else if (generation_ + 1 == generation) {
112 generation_ = generation;
113 old_offset_ = all_offset_;
114 all_offset_ = size;
115 // the generation has been incremented by more than one
116 // (all atoms are marked old)
117 } else if (generation_ + 1 < generation) {
118 generation_ = generation;
119 old_offset_ = size;
120 all_offset_ = size;
121 }
122 }

124 private:
126 size_t old_offset_ = 0;
128 size_t all_offset_ = 0;
130 size_t generation_ = 0;

238 auto add(Symbol atom, StateAtom state) -> AtomUpdate {
239 auto [it, ins] = atoms_.try_emplace(atom, 0, state);
240 if (ins) {
241 if (state != StateAtom::unknown) {
242 derived_.add(atom_index_(it));
243 }
244 return AtomUpdate::added;
245 }
246 auto &prev = it.value();
247 if (state < prev.state) {
248 prev.state = state;
249 if (prev.state == StateAtom::unknown) {
250 derived_.add(atom_index_(it));
251 return AtomUpdate::added;
252 }
253 if (state == StateAtom::fact) {
254 return AtomUpdate::changed;
255 }
256 }
257 return AtomUpdate::unchanged;
258 }
