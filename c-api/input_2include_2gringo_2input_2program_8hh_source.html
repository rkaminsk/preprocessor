150 for (auto const &[id, sym] : const_map_) {
151 fun(Stm{StmConst{sym.first.loc(), sym.first.type(), sym.first.name(),
152 TermSymbol{location(sym.first.value()), sym.second}}});
153 }
154 for (auto const &stm : thy_stms_) {
155 fun(stm);
156 }
157 for (auto const &stm : meta_stms_) {
158 fun(stm);
159 }
160 for (auto const &[sig, part] : parts_) {
161 auto pum = param_map_(store, part);
162 auto loc = part.part.loc();
163 StringVec ids;
164 ids.reserve(sig.second);
165 std::transform(pum.begin(), pum.end(), std::back_inserter(ids), [](auto const &x) { return x.second; });
166 fun(StmProgram{loc, sig.first, std::move(ids)});
167 for (auto const &fact : part.facts) {
168 fun(Stm{StmRule{loc, HdLitSimple{LitSymbolic{loc, Sign::none, TermSymbol{loc, fact}}}, BdLitArray{}}});
169 }
170 for (auto const &stm : part.stms) {
171 if (auto unmapped = unmap_(store, pum, stm); unmapped) {
172 fun(std::move(unmapped).value());
173 } else {
174 fun(stm);
175 }
176 }
177 }
