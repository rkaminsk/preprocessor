85template <auto tag, bool Opt, class Rec, class Arg, class... Args>
86auto select_value(Rec const &rec, Arg &arg, Args &...args) -> decltype(auto) {
87 if constexpr (tag == Arg::tag) {
88 if constexpr (Opt) {
89 using M = std::decay_t<decltype(rec.template get_value<tag>())>;
90 return arg.val ? M{*std::forward<typename Arg::Val>(arg.val)} : M{rec.template get_value<tag>()};
91 } else {
92 return std::forward<typename Arg::Val>(arg.val);
93 }
94 } else if constexpr (sizeof...(args) == 0) {
95 return rec.template get_value<tag>();
96 } else {
97 return select_value<tag, Opt>(rec, args...);
