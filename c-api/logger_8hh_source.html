132inline auto Logger::check(MessageCode code) -> bool {
133 // ignore the message
134 if (static_cast<uint8_t>(code) < static_cast<uint8_t>(level_) || disabled_[static_cast<int>(code)]) {
135 return false;
136 }
137 // report trace and debug messages without reducing the limit
138 if (code < MessageCode::info) {
139 return true;
140 }
141 // report the message
142 if (cur_limit_ > 0) {
143 if (cur_limit_ != std::numeric_limits<size_t>::max()) {
144 --cur_limit_;
145 }
146 return true;
147 }
148 // ignore the message due to limit
149 return false;
150}

152inline auto Logger::enabled(MessageCode code) const -> bool {
